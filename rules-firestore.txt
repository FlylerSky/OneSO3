//rules_r.0.5.3y [Rules 5.4] — account tier management
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // 🔐 HELPER FUNCTIONS
    // Lấy type của caller — 1 get() duy nhất, safe khi field 'type' không tồn tại
    function callerData() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data;
    }

    // Safe read type: trả về 'user' nếu field 'type' không có trong document
    function callerType() {
      return request.auth != null
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        ? ('type' in callerData() ? callerData()['type'] : 'user')
        : 'user';
    }

    // Safe read target type field
    function targetCurrentType() {
      return 'type' in resource.data ? resource.data['type'] : 'user';
    }

    function saTypes() {
      return ['admin','dev','operator','moderator'];
    }

    function isAdmin() {
      return request.auth != null && callerType() == 'admin';
    }

    function isSA() {
      return request.auth != null && callerType() in saTypes();
    }

    // ⚙️ CONFIG/FEATURES
    match /config/{doc} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // 📢 SYSTEM
    match /system/{doc} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // 🗺️ ROADMAP LEAKS
    match /roadmap_leaks/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // 📝 POSTS
    match /posts/{postId} {
      allow read: if true;

      allow create: if
        request.resource.data.createdAt == request.time
        && request.resource.data.likes == 0
        && request.resource.data.dislikes == 0
        && request.resource.data.commentsCount == 0
        && (
          (request.auth != null && request.resource.data.userId == request.auth.uid)
          || (request.auth == null)
        );

      allow delete: if request.auth != null && resource.data.userId == request.auth.uid;

      allow update: if (
        // Nhánh 1: Chủ bài viết chỉnh sửa nội dung
        (request.auth != null
          && resource.data.userId == request.auth.uid
          && request.resource.data.createdAt == resource.data.createdAt
          && request.resource.data.userId   == resource.data.userId
          && !request.resource.data.diff(resource.data).changedKeys().hasAny(['likes','dislikes','commentsCount'])
        )
        ||
        // Nhánh 2: Bất kỳ user cập nhật counters ±1
        (request.auth != null
          && request.resource.data.diff(resource.data).changedKeys().hasOnly(['likes','dislikes','commentsCount'])
          && request.resource.data.likes         is int
          && request.resource.data.dislikes      is int
          && request.resource.data.commentsCount  is int
          && (request.resource.data.likes == resource.data.likes || request.resource.data.likes == resource.data.likes + 1 || request.resource.data.likes == resource.data.likes - 1)
          && (request.resource.data.dislikes == resource.data.dislikes || request.resource.data.dislikes == resource.data.dislikes + 1 || request.resource.data.dislikes == resource.data.dislikes - 1)
          && (request.resource.data.commentsCount == resource.data.commentsCount || request.resource.data.commentsCount == resource.data.commentsCount + 1 || request.resource.data.commentsCount == resource.data.commentsCount - 1)
        )
      );

      // ── COMMENTS
      match /comments/{commentId} {
        allow read: if true;

        allow create: if request.auth != null
          && request.resource.data.userId      == request.auth.uid
          && request.resource.data.displayName  is string
          && request.resource.data.text         is string
          && request.resource.data.createdAt    == request.time;

        allow update: if (
          // Nhánh 1: Chủ comment sửa nội dung
          (request.auth != null
            && resource.data.userId == request.auth.uid
            && request.resource.data.diff(resource.data).changedKeys()
                 .hasOnly(['text','mentions','editedAt','editCount'])
          )
          ||
          // Nhánh 2: Chủ comment sync displayName
          (request.auth != null
            && resource.data.userId == request.auth.uid
            && request.resource.data.displayName is string
            && request.resource.data.diff(resource.data).changedKeys().hasOnly(['displayName'])
          )
          ||
          // Nhánh 3: Chủ comment sync displayName + avatarUrl (profile sync)
          (request.auth != null
            && resource.data.userId == request.auth.uid
            && request.resource.data.displayName is string
            && request.resource.data.diff(resource.data).changedKeys()
                 .hasOnly(['displayName','avatarUrl'])
          )
          ||
          // Nhánh 4: Bất kỳ user tăng reportCount +1
          (request.auth != null
            && request.resource.data.diff(resource.data).changedKeys().hasOnly(['reportCount'])
            && request.resource.data.reportCount is int
            && ((resource.data.keys().hasAll(['reportCount'])
                  && request.resource.data.reportCount == resource.data.reportCount + 1)
                || (!resource.data.keys().hasAll(['reportCount'])
                  && request.resource.data.reportCount == 1))
          )
        );

        allow delete: if request.auth != null && (
          resource.data.userId == request.auth.uid
          || (resource.data.keys().hasAll(['replyTo'])
              && exists(/databases/$(database)/documents/posts/$(postId)/comments/$(resource.data.replyTo))
              && get(/databases/$(database)/documents/posts/$(postId)/comments/$(resource.data.replyTo)).data.userId == request.auth.uid)
        );

        match /reports/{reporterId} {
          allow read: if request.auth != null && request.auth.uid == reporterId;
          allow create: if request.auth != null
            && request.auth.uid == reporterId
            && request.resource.data.userId == request.auth.uid
            && request.resource.data.reason is string
            && request.resource.data.reason in ['spam','offensive','inappropriate','other'];
          allow update, delete: if false;
        }
      }

      // ── LIKES
      match /likes/{likeId} {
        allow read: if true;
        allow create: if request.auth != null
          && request.auth.uid == likeId
          && request.resource.data.userId == request.auth.uid
          && request.resource.data.type in ['like','dislike']
          && request.resource.data.createdAt == request.time;
        allow update: if request.auth != null
          && request.auth.uid == likeId
          && request.resource.data.userId == request.auth.uid
          && request.resource.data.type in ['like','dislike'];
        allow delete: if request.auth != null && request.auth.uid == likeId;
      }
    }

    // 👤 USERS
    match /users/{userId} {
      allow read: if true;

      allow create: if request.auth != null
        && request.auth.uid == userId
        && request.resource.data.email == request.auth.token.email
        && !request.resource.data.keys().hasAny(['type']); // 🔒 không tự gán type

      allow update: if (
        // Nhánh 1: Chính chủ cập nhật profile (mọi field trừ type)
        (request.auth != null
          && request.auth.uid == userId
          && !request.resource.data.diff(resource.data).changedKeys().hasAny(['type']) // 🔒
        )
        ||
        // Nhánh 2: Bất kỳ user cập nhật followersCount/followingCount ±1
        (request.auth != null
          && request.resource.data.diff(resource.data).changedKeys()
               .hasOnly(['followersCount','followingCount'])
          && ((resource.data.followersCount is int
                && (request.resource.data.followersCount == resource.data.followersCount
                    || request.resource.data.followersCount == resource.data.followersCount + 1
                    || request.resource.data.followersCount == resource.data.followersCount - 1))
              || (!resource.data.keys().hasAll(['followersCount'])
                && request.resource.data.followersCount == 1))
          && ((resource.data.followingCount is int
                && (request.resource.data.followingCount == resource.data.followingCount
                    || request.resource.data.followingCount == resource.data.followingCount + 1
                    || request.resource.data.followingCount == resource.data.followingCount - 1))
              || (!resource.data.keys().hasAll(['followingCount'])
                && request.resource.data.followingCount == 1))
        )
        ||
        // Nhánh 3: SA cập nhật type của user KHÁC (cross-user type management)
        // - Admin: quyền cao nhất, đổi được bất kỳ ai sang bất kỳ type
        // - Dev/Operator/Moderator: chỉ đổi được RA/LLA → RA/LLA (không đụng SA)
        (request.auth != null
          && request.auth.uid != userId
          && isSA()
          && request.resource.data.diff(resource.data).changedKeys().hasOnly(['type'])
          && request.resource.data.type is string
          && (
               callerType() == 'admin'
               ||
               (
                 callerType() in saTypes()
                 && !(targetCurrentType() in saTypes())
                 && !(request.resource.data.type in saTypes())
               )
             )
        )
        ||
        // Nhánh 4: Admin hạ cấp chính mình khi chuyển quyền
        // Chỉ cho phép admin → SA khác (không xuống RA/LLA để bảo toàn hệ thống)
        (request.auth != null
          && request.auth.uid == userId
          && targetCurrentType() == 'admin'
          && request.resource.data.diff(resource.data).changedKeys().hasOnly(['type'])
          && request.resource.data.type is string
          && request.resource.data.type in saTypes()
          && request.resource.data.type != 'admin'
        )
      );

      allow delete: if request.auth != null && request.auth.uid == userId;

      match /searchHistory/{doc} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }

      match /followers/{followerId} {
        allow read: if true;
        allow create: if request.auth != null
          && request.auth.uid == followerId
          && request.resource.data.userId    == request.auth.uid
          && request.resource.data.createdAt == request.time;
        allow delete: if request.auth != null && request.auth.uid == followerId;
        allow update: if false;
      }

      match /following/{followingId} {
        allow read: if true;
        allow create: if request.auth != null
          && request.auth.uid == userId
          && request.resource.data.userId    == followingId
          && request.resource.data.createdAt == request.time;
        allow delete: if request.auth != null && request.auth.uid == userId;
        allow update: if false;
      }

      match /visitors/{visitorId} {
        allow read: if request.auth != null && request.auth.uid == userId;
        allow create, update: if request.auth != null
          && request.auth.uid == visitorId
          && request.resource.data.userId         == request.auth.uid
          && request.resource.data.lastVisitedAt  == request.time;
        allow delete: if request.auth != null
          && (request.auth.uid == userId || request.auth.uid == visitorId);
      }
    }

    // 🔍 SEARCH STATS
    match /searchStats/{doc} {
      allow read: if true;
      allow write: if true;
    }

    // 🔔 NOTIFICATIONS
    match /notifications/{notificationId} {
      allow read: if true;
      allow write: if isAdmin();
    }

  }
}
